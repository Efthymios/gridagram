Gridagram: Example 2

Simple data array with unusual content fields


        var data = [ 
            { a: "a-title", b: { c: "a-body" } },
            { a: "b-title", b: { c: "b-body" } },
            { a: "c-title", b: { c: "c-body" } }
        ];

        $("#results").gridagram(data, { 
            titleField: "a", 
            bodyField: function(e) { return e.b.c; }
        });
